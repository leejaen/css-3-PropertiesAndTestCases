CSS3 Transitions过渡事件响应

语法：

事件webkitTransitionEnd

实例

        var timeOutId=0;
        document.getElementById("move").addEventListener("webkitTransitionEnd",function()
        {
            document.getElementById("tip").innerHTML="过度结束 收到webkitTransitionEnd事件  当前事件对象为'"+event.target.id+"'"+"--时间:"+(new Date()).getTime();
            
            clearTimeout(timeOutId);
            timeOutId=setTimeout(function(){document.getElementById("tip").innerHTML=" "},4000);
        },true);
